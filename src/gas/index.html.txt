  <script>
    let currentData = {};
    let currentStep = 'select_type';
    const STEPS = ['select_type', 'secret_check', 'form_general', 'form_alumni', 'review', 'submitting', 'success'];

    function showTab(tabId) {
      document.getElementById('formTab').classList.add('hidden');
      document.getElementById('dashboardTab').classList.add('hidden');
      document.getElementById('btn-formTab').className = "text-gray-500 hover:text-gray-700 px-3 py-2 font-medium text-sm";
      document.getElementById('btn-dashboardTab').className = "text-gray-500 hover:text-gray-700 px-3 py-2 font-medium text-sm";
      document.getElementById(tabId).classList.remove('hidden');
      document.getElementById('btn-' + tabId).className = "text-blue-600 border-b-2 border-blue-600 px-3 py-2 font-medium text-sm";
      if(tabId === 'dashboardTab') loadDashboard();
    }

    function goToStep(stepId) {
      STEPS.forEach(s => {
        const el = document.getElementById('step-' + s);
        if(el) el.classList.add('hidden');
      });
      document.getElementById('step-' + stepId).classList.remove('hidden');
      if (stepId === 'submitting') {
        document.getElementById('step-' + stepId).classList.add('flex');
      }
      if (stepId === 'success') {
        document.getElementById('step-' + stepId).classList.add('flex');
      }
      currentStep = stepId;
    }

    function handleTypeChange(type) {
      currentData.userType = type;
      if (type === 'general') {
        goToStep('form_general');
      } else {
        goToStep('secret_check');
      }
    }

    function verifySecret() {
      const code = document.getElementById('secretCode').value;
      const teacher = document.getElementById('referringTeacher').value;
      if (code.trim() === 'NPTU-MENTOR26' && teacher.trim() !== '') {
        currentData.secretCode = code;
        currentData.referringTeacher = teacher;
        goToStep('form_alumni');
      } else {
        alert('รหัสลับไม่ถูกต้อง หรือ ลืมกรอกชื่ออาจารย์ผู้แนะนำ');
      }
    }

    function submitFormGeneral(e) {
      e.preventDefault();
      const fileInput = document.getElementById('g_slipFile');
      if(fileInput.files.length === 0) { alert("กรุณาแนบไฟล์"); return; }
      currentData.name = document.getElementById('g_name').value;
      currentData.phone = document.getElementById('g_phone').value;
      currentData.email = document.getElementById('g_email').value;
      currentData.file = fileInput.files[0];
      buildReview();
      goToStep('review');
    }

    function submitFormAlumni(e) {
      e.preventDefault();
      currentData.name = document.getElementById('a_name').value;
      currentData.phone = document.getElementById('a_phone').value;
      currentData.email = document.getElementById('a_email').value;
      currentData.lineId = document.getElementById('a_line').value;
      currentData.licenseNumber = document.getElementById('a_licenseNo').value;
      currentData.licenseExpiry = document.getElementById('a_licenseExp').value;
      buildReview();
      goToStep('review');
    }

    function buildReview() {
      let html = '';
      const typeLabel = currentData.userType === 'general' ? 'บุคคลทั่วไป' : 'ศิษย์เก่า/พยาบาลพี่เลี้ยง';
      html += `<div class="sm:col-span-2"><dt class="text-sm font-medium text-gray-500">ประเภทผู้สมัคร</dt><dd class="mt-1 text-sm font-semibold text-gray-900">${typeLabel}</dd></div>`;
      html += `<div><dt class="text-sm font-medium text-gray-500">ชื่อ-นามสกุล</dt><dd class="mt-1 text-sm text-gray-900">${currentData.name}</dd></div>`;
      html += `<div><dt class="text-sm font-medium text-gray-500">เบอร์โทรศัพท์</dt><dd class="mt-1 text-sm text-gray-900">${currentData.phone}</dd></div>`;
      html += `<div><dt class="text-sm font-medium text-gray-500">อีเมล</dt><dd class="mt-1 text-sm text-gray-900">${currentData.email}</dd></div>`;
      
      if (currentData.userType === 'general') {
        html += `<div class="sm:col-span-2"><dt class="text-sm font-medium text-gray-500">ไฟล์หลักฐาน (ชื่อไฟล์)</dt><dd class="mt-1 text-sm text-gray-900">${currentData.file.name}</dd></div>`;
      } else {
        html += `<div><dt class="text-sm font-medium text-gray-500">อาจารย์ผู้แนะนำ</dt><dd class="mt-1 text-sm text-gray-900">${currentData.referringTeacher}</dd></div>`;
        html += `<div><dt class="text-sm font-medium text-gray-500">ID Line</dt><dd class="mt-1 text-sm text-gray-900">${currentData.lineId}</dd></div>`;
        html += `<div><dt class="text-sm font-medium text-gray-500">เลขที่ใบประกอบวิชาชีพ</dt><dd class="mt-1 text-sm text-gray-900">${currentData.licenseNumber}</dd></div>`;
        html += `<div><dt class="text-sm font-medium text-gray-500">วันหมดอายุใบประกอบวิชาชีพ</dt><dd class="mt-1 text-sm text-gray-900">${currentData.licenseExpiry}</dd></div>`;
      }
      document.getElementById('review-content').innerHTML = html;
    }

    function backToForm() {
      goToStep(currentData.userType === 'general' ? 'form_general' : 'form_alumni');
    }

    function submitData() {
      goToStep('submitting');
      if (currentData.userType === 'general' && currentData.file) {
        const reader = new FileReader();
        reader.onload = function(e) {
          const base64Data = e.target.result.split(',')[1];
          callGAS({
            ...currentData,
            slipFile: { filename: currentData.file.name, mimeType: currentData.file.type, data: base64Data }
          });
        };
        reader.readAsDataURL(currentData.file);
      } else {
        callGAS(currentData);
      }
    }

    function callGAS(payload) {
      google.script.run
        .withSuccessHandler(function(response) {
          if(response.success) { goToStep('success'); } 
          else { alert("Error: " + response.message); backToForm(); }
        })
        .withFailureHandler(function(error) {
          alert("Connection Failed: " + error); backToForm();
        })
        .submitRegistration(payload);
    }
    
    function loadDashboard() {
      document.getElementById('dash-loading').classList.remove('hidden');
      document.getElementById('dash-content').classList.add('hidden');
      google.script.run
        .withSuccessHandler(function(res) {
          document.getElementById('dash-loading').classList.add('hidden');
          if(res.success) {
            document.getElementById('dash-content').classList.remove('hidden');
            document.getElementById('dash-total').innerText = res.data.total;
            document.getElementById('dash-revenue').innerText = res.data.revenue.toLocaleString() + " บาท";
            document.getElementById('dash-general').innerText = res.data.general;
            document.getElementById('dash-alumni').innerText = res.data.alumni;
          }
        })
        .getDashboardData();
    }
  </script>
